{

	//$Arg0Type 11
	//$Arg0Enum { 0 = "No"; 1 = "Yes"; }

	Scale 0.6
	RenderStyle Translucent	
	+NOBLOCKMAP
	+MOVEWITHSECTOR

	{
	Spawn:
		FWAT A 0 NoDelay A_JumpIf (args[0] == 1, "Sound")

		FWAT A 0 A_StartSound("Fountain/Loop", CHAN_6, CHANF_LOOP, frandom(0.7,1.0), ATTN_IDLE) //randomized volume this time - ozy81
	Animate:
		FWAT ABCD 3
		Loop
	}
}
