Actor Sparks 11914
{
	Tag "Spark"
	Health 50
	Radius 18
	Height 36
	Mass 20000
	Speed 12
	PainChance 0
	SeeSound "SPKSee"
  PainSound "SPKPain"
	HitObituary "$HB_SPARK"
	MONSTER
	+NOGRAVITY
	+FLOAT 
	 +NOBLOOD
	 +NOINFIGHTING
	 +DONTHARMSPECIES
	 +COUNTKILL
	States
	{
	Spawn:
		LSPK AB 1 A_Look
		Loop
    See:
		TNT1 A 0 A_JumpIfInTargetLOS("Seen", 130)
		LSPK ABABABABABAB 1 BRIGHT A_Chase
		Loop
    Seen:
		TNT1 A 0
        TNT1 A 0
		LSPK ABABABABABAB 1 
		TNT1 A 0 A_JumpIfInTargetLOS("Seen", 130)
		Goto See
	Melee:
		LSPK ABABABEF 1 A_FaceTarget
		LSPK E 1 A_CustomMeleeAttack(random (2, 3) *5, "SPKBUZZ", "SPKBUZZ")
		LSPK FEFABAB 1
		Goto See
	Death:
		TNT1 AAAAAAA 0 A_SpawnItemEx("SparkPiece",0,0,random(8,64),frandom(-5,5),frandom(-5,5),frandom(2,9))
		TNT1 A 0 A_GiveToTarget("Points",100)
		Stop
	}
}
