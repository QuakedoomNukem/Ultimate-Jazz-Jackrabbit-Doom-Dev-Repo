Actor JetpackDoofus 3235
{
	//$Category JJ_Goons
	Tag "Jetpack Doofus"
	Health 150
	Radius 12
	Height 40
	Mass 0x7fffffff
	Speed 6
	PainChance 0
	DamageFactor "JZombie", 0
	MONSTER
	+COUNTKILL
	+DONTFALL
	+DONTHARMCLASS
	+FLOAT
	+FLOATBOB
	+FLOORCLIP
	+NOBLOOD
	+NOGRAVITY
	+NOINFIGHTING
	SeeSound "TURTSIGHT"
	PainSound "TURTPAIN"
	ActiveSound "TURTACT"
	Obituary "$OB_JDOOF"
	States
	{
	Spawn:
		JETT A 0
		JETT A 0 A_PlaySoundEx("Hover2", "SoundSlot7", 1)
		JETT A 1 A_Look
		Loop
	See:
		JETT A 0 A_PlaySoundEx("Hover2", "SoundSlot7", 1)
		JETT A 2 A_Chase
		Loop
		JETT A 0 A_PlaySoundEx("Hover2", "SoundSlot7", 1)
		JETT A 0 A_FaceTarget
		JETT A 0 A_Stop
		Goto See+2
	Missile:
		JETT AAA 20 A_FaceTarget
		JETT A 0 A_PlaySoundEx("Hover2", "SoundSlot7", 1)
		JETT C 6 BRIGHT A_CustomMissile("TurtleGoonBullet",32,0,0,0,0)
		Goto See
	Death:
		TNT1 A 0 A_StopSound(CHAN_VOICE)
		TNT1 AAAAAAAAAA 0 A_SpawnItemEx("JazzPoof",0,0,24,frandom(-8,8),frandom(-8,8),frandom(4,8))
		TNT1 A 0 A_GiveToTarget("Points",150)
		Stop
	Death.JazzprojIce:
		TNT1 A 0 A_GiveToTarget("Points",150)
		JETT A 5 A_GenericFreezeDeath
		JETT A 1 A_FreezeDeathChunks
		Wait
	Death.JazzprojFire:
	Burn:
		TNT1 A 0 A_NoBlocking
		TNT1 A 0 A_GiveToTarget("Points",150)
		TNT1 A 0 A_PlaySound("FMTRFIRE")
		JETT AAA 2 A_SpawnItemEx("FireDeadSpawner",0,0,0,0,0,0,0,128)
		TNT1 A 2 A_SpawnItemEx("FireDead",0,0,0,0,0,0,0,128)
		TNT1 A 0 A_SpawnItemEx("AshHolm",0,0,0,0,0,0,0,128)
		Stop
	}
}
