Actor XmasSnowTurtle1 4094
{
  Tag "Snow Doofus"
  Health 50
  Radius 20
  Height 56
  Speed 15
  PainChance 200
  MONSTER
  +FLOORCLIP
  +NOBLOOD
  +NOINFIGHTING
  SeeSound "Turtsight"
  PainSound "Turtpain"
  ActiveSound "Turtact"
  Obituary "$OB_XSTUR"
  States
  {
  Spawn:
    SNTL AB 10 A_Look
    Loop
  See:
    SNTL AABBCCDD 4 A_Chase
    Loop
  Missile:
    SNTL E 10 A_FaceTarget
    SNTL F 8 A_CustomMissile("Snowball",32,0,0,0,0)
    SNTL E 8
    Goto See
  Pain:
    SNTL G 3
    SNTL G 3 A_Pain
    Goto See
   Death:
    TNT1 AAAAAAAAAA 0 A_SpawnItemEx("JazzPoof",0,0,24,frandom(-8,8),frandom(-8,8),frandom(4,8))
	TNT1 A 0 A_GiveToTarget("Points",100)
    Stop
	Death.JazzprojIce:
	TNT1 A 0 A_GiveToTarget("Points",100)
	SNTL A 5 A_GenericFreezeDeath
    SNTL A 1 A_FreezeDeathChunks
    Wait
	Death.JazzprojFire:
	Burn:
	TNT1 A 0 A_NoBlocking
	TNT1 A 0 A_GiveToTarget("Points",100)
	TNT1 A 0 A_StartSound("FMTRFIRE")
	SNTL A 2 A_SpawnItemEx("FireDead",0,0,0,0,0,0,0,128)
	TNT1 A 0 A_SpawnItemEx("AshHolm",0,0,0,0,0,0,0,128)
	Stop
  }
}
