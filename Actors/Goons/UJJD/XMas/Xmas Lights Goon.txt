Actor XmasLightsGoon 4242
{
	Tag "XMas Lights Goon"
	Health 150
	Radius 16
	Height 32
	Speed 10
	PainChance 0
	Mass 400
	+FLOORCLIP
	+NOBLOOD
	+NOINFIGHTING
	  +COUNTKILL
+DONTHARMSPECIES
	MONSTER
	SeeSound "Turtsight"
  PainSound "Turtpain"
  ActiveSound "Turtact"
	AttackSound "TURTBITE"
	Obituary "$OB_MGOON"
	States
	{
		Spawn:
		    xmst AB 10 A_Look
			Loop
		See:
			xmst AABBCCDDEEFF 2 Fast A_Chase
			Loop
		Melee:
			xmst GH 6 A_FaceTarget
			TNT1 A 0 A_StartSound("TURTBITE")
			xmst I 6 A_SargAttack
			Goto See
		Death:
		    TNT1 AAAAAAAAAA 0 A_SpawnItemEx("JazzPoof",0,0,24,frandom(-8,8),frandom(-8,8),frandom(4,8))
           TNT1 A 0 A_GiveToTarget("Points",150)
		   Stop
		   Death.JazzprojIce:
	TNT1 A 0 A_GiveToTarget("Points",150)
	xmst A 5 A_GenericFreezeDeath
    xmst A 1 A_FreezeDeathChunks
    Wait
    Death.JazzprojFire:
	Burn:
	TNT1 A 0 A_NoBlocking
	TNT1 A 0 A_GiveToTarget("Points",150)
	TNT1 A 0 A_StartSound("FMTRFIRE")
	xmst AAA 2 A_SpawnItemEx("FireDeadSpawner",0,0,0,0,0,0,0,128)
	TNT1 A 2 A_SpawnItemEx("FireDead",0,0,0,0,0,0,0,128)
	TNT1 A 0 A_SpawnItemEx("AshHolm",0,0,0,0,0,0,0,128)
	Stop
	}
}
