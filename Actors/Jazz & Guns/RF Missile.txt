Actor DualRocket
{
	Radius 11
	Height 4
	Speed 50
	Damage 50
	DamageType JazzProj
	ReactionTime 20
	Projectile
	+BRIGHT
	+DEHEXPLOSION
	+RANDOMIZE
	+RANDOMIZE
	+ROCKETTRAIL
	SeeSound "RFLAUNCH"
	DeathSound "SEEKXPLD"
	Obituary "$OB_DUALR"
	States
	{
	Spawn:
		RFLM A 1
		RFLM A 0 A_Countdown
		Loop
	Death:
		RFLE AB 3
		RFLE C 3 A_Explode (128,128,0)
		RFLE DEFGHIJK 3
		Stop
	}
}
