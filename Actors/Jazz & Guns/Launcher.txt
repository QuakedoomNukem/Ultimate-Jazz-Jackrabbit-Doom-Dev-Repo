Actor 40mmShot
{
	Radius 8
	Height 8
	Damage 200
	Speed 30
	Scale 0.75
	BounceType "Hexen"
	BounceCount 20
	BounceFactor 1.0
	DamageType JazzProj
	Projectile
	-NOGRAVITY
	+SEEKERMISSILE
	+USEBOUNCESTATE
	BounceSound "BOUNCE"
	DeathSound "BLSTRHIT"
	States
	{
	Spawn:
		SHRP O 3
		SHRP O 0 A_CountDown
		Loop
	Death:
	    JBLS EFGH 3
		Stop
	}
}
